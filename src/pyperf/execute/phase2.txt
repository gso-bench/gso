#!/bin/bash
set -e

echo "Phase 2: Running performance comparison..."

# Read the working pair
read working_commit test_file < working_pair.txt
echo "Using commit: $${working_commit}, test: $${test_file}"

cd $repo_name
echo "Testing base commit: $${working_commit}^"
git checkout "$${working_commit}^"
$install_commands

cd ..
echo "Running tests with working candidate commit..."
for i in {1..5}; do
    python working_test.py ${file_before}
done

cd $repo_name
echo "Testing target commit: ${target_commit}"
git checkout ${target_commit}
$install_commands

cd ..
echo "Running tests with target commit..."
for i in {1..5}; do
    python working_test.py ${file_after}
done

echo "Task complete on $(hostname)"
